\section{Question 2: Vectors Similarity}

\textbf{Given movie embedding vectors:}
\begin{itemize}
    \item Movie A: ``Laser Quest'' (Sci-Fi Action): $\mathbf{a} = (1, 2, 3)$
    \item Movie B: ``Sunny Days'' (Family Comedy): $\mathbf{b} = (2, 1, 0)$
    \item Movie C: ``Nebula Dreams'' (Sci-Fi Drama): $\mathbf{c} = (1, 2, 2)$
\end{itemize}

\subsection{Question 2(a)}

\textbf{Compute the similarities or distances for the following queries.}

\subsubsection*{(i) Cosine Similarity: ``Laser Quest'' to ``Nebula Dreams''}

Formula: $\cos\_sim(\mathbf{x}, \mathbf{y}) = \frac{\mathbf{x} \cdot \mathbf{y}}{\|\mathbf{x}\| \|\mathbf{y}\|}$

\textbf{Step 1: Compute the dot product $\mathbf{a} \cdot \mathbf{c}$}
\begin{align*}
\mathbf{a} \cdot \mathbf{c} &= (1)(1) + (2)(2) + (3)(2) \\
&= 1 + 4 + 6 \\
&= 11
\end{align*}

\textbf{Step 2: Compute the norms}
\begin{align*}
\|\mathbf{a}\| &= \sqrt{1^2 + 2^2 + 3^2} = \sqrt{1 + 4 + 9} = \sqrt{14} \\
\|\mathbf{c}\| &= \sqrt{1^2 + 2^2 + 2^2} = \sqrt{1 + 4 + 4} = \sqrt{9} = 3
\end{align*}

\textbf{Step 3: Compute cosine similarity}
\begin{align*}
\cos\_sim(\mathbf{a}, \mathbf{c}) &= \frac{11}{\sqrt{14} \times 3} = \frac{11}{3\sqrt{14}} = \frac{11}{11.225} \approx \boxed{0.980}
\end{align*}

\textit{Interpretation:} A cosine similarity of 0.980 (very close to 1) indicates that ``Laser Quest'' and ``Nebula Dreams'' are highly similar in their embedding space, which makes sense as both are Sci-Fi movies.

\vspace{1em}

\subsubsection*{(ii) Euclidean Distance: ``Laser Quest'' to ``Sunny Days''}

Formula: $d_E(\mathbf{x}, \mathbf{y}) = \sqrt{\sum_i (x_i - y_i)^2}$

\begin{align*}
d_E(\mathbf{a}, \mathbf{b}) &= \sqrt{(1-2)^2 + (2-1)^2 + (3-0)^2} \\
&= \sqrt{(-1)^2 + (1)^2 + (3)^2} \\
&= \sqrt{1 + 1 + 9} \\
&= \sqrt{11} \\
&\approx \boxed{3.317}
\end{align*}

\textit{Interpretation:} The Euclidean distance of $\sqrt{11} \approx 3.317$ indicates a moderate distance between ``Laser Quest'' (Sci-Fi Action) and ``Sunny Days'' (Family Comedy), reflecting their different genres.

\vspace{1em}

\subsubsection*{(iii) Manhattan Distance: ``Laser Quest'' to ``Nebula Dreams''}

Formula: $d_{L1}(\mathbf{x}, \mathbf{y}) = \sum_i |x_i - y_i|$

\begin{align*}
d_{L1}(\mathbf{a}, \mathbf{c}) &= |1-1| + |2-2| + |3-2| \\
&= 0 + 0 + 1 \\
&= \boxed{1}
\end{align*}

\textit{Interpretation:} A Manhattan distance of only 1 confirms that ``Laser Quest'' and ``Nebula Dreams'' are very close in the embedding space, differing only in the third dimension (action level: 3 vs 2).

\vspace{1em}

\subsubsection*{(iv) Dot Product: ``Sunny Days'' and ``Nebula Dreams''}

Formula: $\text{dot}(\mathbf{x}, \mathbf{y}) = \sum_i x_i y_i$

\begin{align*}
\text{dot}(\mathbf{b}, \mathbf{c}) &= (2)(1) + (1)(2) + (0)(2) \\
&= 2 + 2 + 0 \\
&= \boxed{4}
\end{align*}

\textit{Interpretation:} The dot product of 4 measures the alignment of the two vectors. A positive value indicates some thematic alignment between ``Sunny Days'' and ``Nebula Dreams'', though not as strong as between the two Sci-Fi movies.

\textbf{New user X has embedding: $\mathbf{X} = (3, 1)$}

\subsubsection*{Step 1: Calculate Euclidean Distance from X to Each User}

Using the formula: $d_E(\mathbf{X}, \mathbf{p}) = \sqrt{(x_1 - p_1)^2 + (x_2 - p_2)^2}$

\begin{align*}
d(X, A) &= \sqrt{(3-1)^2 + (1-2)^2} = \sqrt{4 + 1} = \sqrt{5} \approx 2.236 \\
d(X, B) &= \sqrt{(3-2)^2 + (1-4)^2} = \sqrt{1 + 9} = \sqrt{10} \approx 3.162 \\
d(X, C) &= \sqrt{(3-3)^2 + (1-3)^2} = \sqrt{0 + 4} = \sqrt{4} = 2 \\
d(X, D) &= \sqrt{(3-6)^2 + (1-5)^2} = \sqrt{9 + 16} = \sqrt{25} = 5 \\
d(X, E) &= \sqrt{(3-7)^2 + (1-3)^2} = \sqrt{16 + 4} = \sqrt{20} \approx 4.472
\end{align*}

\subsubsection*{Step 2: Sort Users by Distance and Find 3 Nearest Neighbors}

\begin{center}
\begin{tabular}{|c|c|c|c|}

\hline
4 & E & $\sqrt{20} \approx 4.472$ & Drama \\
\hline
5 & D & $\sqrt{25} = 5$ & Drama \\
\hline
\end{tabular}

\textbf{3 Nearest Neighbors:} C, A, B

\subsubsection*{Step 3: Majority Vote for Genre Prediction}

\textit{Explanation:} The two closest users to X (users C and A) both prefer Sci-Fi, while only one neighbor (B) prefers Comedy. Therefore, by majority vote (2 vs 1), we predict that user X will prefer Sci-Fi content.

\subsection{Question 3(a)}

\textbf{IVF Index with Two Coarse Centroids}

\textbf{Given centroids:}
\begin{itemize}
    \item $c_1 = (1.5, 1.5)$ --- Low-intensity / lightweight movies
    \item $c_2 = (6.0, 5.5)$ --- High-intensity / epic movies
\end{itemize}

\textbf{Assign each movie to its nearest centroid using Euclidean distance.}

\subsubsection*{Distance Calculations}

\textbf{Movie A: Laser Quest (1, 2)}
\begin{align*}
d(A, c_1) &= \sqrt{(1-1.5)^2 + (2-1.5)^2} = \sqrt{0.25 + 0.25} = \sqrt{0.5} \approx 0.707 \\
d(A, c_2) &= \sqrt{(1-6)^2 + (2-5.5)^2} = \sqrt{25 + 12.25} = \sqrt{37.25} \approx 6.103
\end{align*}
\textit{Nearest centroid:} $c_1$ \checkmark

\textbf{Movie B: Sunny Days (2, 1)}
\begin{align*}
d(B, c_1) &= \sqrt{(2-1.5)^2 + (1-1.5)^2} = \sqrt{0.25 + 0.25} = \sqrt{0.5} \approx 0.707 \\
d(B, c_2) &= \sqrt{(2-6)^2 + (1-5.5)^2} = \sqrt{16 + 20.25} = \sqrt{36.25} \approx 6.021
\end{align*}
\textit{Nearest centroid:} $c_1$ \checkmark

\textbf{Movie C: Nebula Dreams (4, 5)}
\begin{align*}
d(C, c_1) &= \sqrt{(4-1.5)^2 + (5-1.5)^2} = \sqrt{6.25 + 12.25} = \sqrt{18.5} \approx 4.301 \\
d(C, c_2) &= \sqrt{(4-6)^2 + (5-5.5)^2} = \sqrt{4 + 0.25} = \sqrt{4.25} \approx 2.062
\end{align*}
\textit{Nearest centroid:} $c_2$ \checkmark

\textbf{Movie D: Shadow Empire (7, 6)}
\begin{align*}
d(D, c_1) &= \sqrt{(7-1.5)^2 + (6-1.5)^2} = \sqrt{30.25 + 20.25} = \sqrt{50.5} \approx 7.106 \\
d(D, c_2) &= \sqrt{(7-6)^2 + (6-5.5)^2} = \sqrt{1 + 0.25} = \sqrt{1.25} \approx 1.118
\end{align*}
\textit{Nearest centroid:} $c_2$ \checkmark

\subsubsection*{Resulting Clusters}

\begin{center}
\begin{tabular}{|c|c|l|}
\hline
\textbf{Cluster} & \textbf{Centroid} & \textbf{Movies} \\

\subsection{Question 3(b)}

\textbf{IVF Search for Nearest Neighbor of Q}

\textbf{Query:} New movie $\mathbf{Q} = (3, 3.5)$

\subsubsection*{Step 1: Find Nearest Centroid to Q}

\begin{align*}
d(Q, c_1) &= \sqrt{(3-1.5)^2 + (3.5-1.5)^2} = \sqrt{2.25 + 4} = \sqrt{6.25} = 2.5 \\
d(Q, c_2) &= \sqrt{(3-6)^2 + (3.5-5.5)^2} = \sqrt{9 + 4} = \sqrt{13} \approx 3.606
\end{align*}

\textbf{Nearest centroid:} $c_1$ (distance = 2.5)

\subsubsection*{Step 2: Search Only in Cluster 1 (One Probe)}

Cluster 1 contains: Laser Quest (A) at (1, 2) and Sunny Days (B) at (2, 1)

\textbf{Calculate distances from Q to movies in Cluster 1:}
\begin{align*}
d(Q, A) &= \sqrt{(3-1)^2 + (3.5-2)^2} = \sqrt{4 + 2.25} = \sqrt{6.25} = 2.5 \\
d(Q, B) &= \sqrt{(3-2)^2 + (3.5-1)^2} = \sqrt{1 + 6.25} = \sqrt{7.25} \approx 2.693
\end{align*}

\subsubsection*{IVF Search Result}

$$\boxed{\text{Nearest Neighbor (IVF): \textbf{Laser Quest (A)} with distance } 2.5}$$




\subsection{Question 3(c)}

\textbf{Brute Force Verification: Is the IVF Answer the True Nearest Neighbor?}

\subsubsection*{Compute Distance from Q to ALL Movies}

\begin{align*}
d(Q, A) &= \sqrt{(3-1)^2 + (3.5-2)^2} = \sqrt{4 + 2.25} = \sqrt{6.25} = 2.5 \\
d(Q, B) &= \sqrt{(3-2)^2 + (3.5-1)^2} = \sqrt{1 + 6.25} = \sqrt{7.25} \approx 2.693 \\
d(Q, C) &= \sqrt{(3-4)^2 + (3.5-5)^2} = \sqrt{1 + 2.25} = \sqrt{3.25} \approx 1.803 \\
d(Q, D) &= \sqrt{(3-7)^2 + (3.5-6)^2} = \sqrt{16 + 6.25} = \sqrt{22.25} \approx 4.717
\end{align*}

\subsubsection*{Ranked List (Brute Force)}

\begin{center}
\begin{tabular}{|c|c|c|c|}
\hline
\textbf{Rank} & \textbf{Movie} & \textbf{Distance to Q} & \textbf{Note} \\
\hline
1 & Nebula Dreams (C) & $\sqrt{3.25} \approx 1.803$ & \textbf{TRUE Nearest Neighbor} \\
\hline
2 & Laser Quest (A) & $\sqrt{6.25} = 2.5$ & IVF Answer \\
\hline
3 & Sunny Days (B) & $\sqrt{7.25} \approx 2.693$ & \\
\hline
4 & Shadow Empire (D) & $\sqrt{22.25} \approx 4.717$ & \\
\hline
\end{tabular}
\end{center}

\subsubsection*{Conclusion}

$$\boxed{\text{The IVF answer is \textbf{NOT} the true nearest neighbor.}}$$

\textbf{Analysis:}
\begin{itemize}
    \item \textbf{True NN:} Nebula Dreams (C) with distance $\approx 1.803$
    \item \textbf{IVF Answer:} Laser Quest (A) with distance $= 2.5$
    \item \textbf{IVF Rank:} The IVF answer is the \textbf{2nd closest} movie (not the 1st)
\end{itemize}

\textbf{Why did IVF miss the true NN?}

The true nearest neighbor (Nebula Dreams) is in \textbf{Cluster 2}, but the IVF search only probed \textbf{Cluster 1} because Q was closer to centroid $c_1$. This demonstrates the fundamental trade-off of IVF indexing:

\begin{itemize}
    \item \textbf{Advantage:} Faster search (examined only 2 movies instead of 4)
    \item \textbf{Disadvantage:} May return approximate results, missing the true NN if it lies in a different cluster
\end{itemize}
